
\textbf{NORMĀLAIS SADALĪJUMS}

\equationEntry{Sadalījuma aprēkināšana:}
{\begin{align*}
    x = NORM.DIST(X, \mu, \sigma, CUMM) \\
\end{align*}}


\equationEntry{Vērtības un $\sigma$ iegūšana no varbūtības:}
{\begin{align*}
    Z &= NORM.S.INV(x) \\
    \sigma &= \displaystyle\frac{x-\mu}{Z} 
\end{align*}}


\equationEntry{Sadalījuma aprēkināšana:}
{\begin{align*}
    x = NORM.DIST(X, \mu, \sigma, CUMM) \\
\end{align*}}


\equationEntry{Vērtības un $\sigma$ iegūšana no varbūtības:}
{\begin{align*}
    Z &= NORM.S.INV(x) \\
    \sigma &= \displaystyle\frac{x-\mu}{Z} 
\end{align*}}


\equationEntry{Empīriskie likumi:}
{\begin{align*}
    \mu \pm 1\sigma &= 68,26\%  \\
    \mu \pm 2\sigma &= 95,44\% \\
    \mu \pm 3\sigma &= 99,73\% \\
\end{align*}}
